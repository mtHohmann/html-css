O código abaixo é escrito em C++ e serve para caulcuar a media de alunos:

        
    #include 
        using namespace std;

        class Math
        {
        public:
            Math(double, double);
            Math(double, double, double);
        };

        Math::Math(double n1, double n2)
        {
            cout << "Media: " << (n1 + n2) / 2 << endl;
        }

        Math::Math(double n1, double n2, double n3)
        {
            cout << "Media: " << (n1 + n2 + n3) / 3 << endl;
        }

        int main()
        {
            Math Be(9.5, 7.4);
            Math Annie(8.5, 8.9);

            return 0;
        }
